```typescript
// src/lib/shiwakechou/parsers/bank/rakuten.ts (新規 or spec 既存)
export function parseRakutenCsv(buffer: Buffer): RakutenTransaction[] {
  // 1. Shift-JIS → UTF-8 変換 (iconv-lite 等)
  const text = decodeSjis(buffer);
  
  // 2. 行分割 (CRLF)
  const lines = text.split(/\r?\n/).filter(l => l.length > 0);
  
  // 3. ヘッダー行スキップ (lines[0] = "取引日,入出金額(円),残高(円),取引内容")
  const dataLines = lines.slice(1);
  
  // 4. 各行 4 列パース
  return dataLines.map(line => {
    const [dateStr, amountStr, balanceStr, description] = line.split(',');
    return {
      transaction_date: parseDate(dateStr), // YYYYMMDD → Date
      amount: parseInt(amountStr),          // 負=出金、正=入金
      balance: parseInt(balanceStr),        // 取引後残高
      description,
      flow: parseInt(amountStr) < 0 ? 'withdrawal' : 'deposit',
    };
  });
}
```
